<script>
  // ==========================================
  // 交互逻辑（纯前端，请勿移除）
  // 1. 必须滚动到底部才能启用勾选
  // 2. 同意状态存入 localStorage，下次访问无需重新确认
  // ==========================================
  (function() {
    const modal = document.getElementById('agreement-modal');
    const checkbox = document.getElementById('agree-checkbox');
    const confirmBtn = document.getElementById('confirm-btn');

    // 如果之前已同意过，直接跳过弹窗
    if (localStorage.getItem('muse-tool-agreed') === 'yes') {
      modal.style.display = 'none';
      return;
    }

    // 检测是否滚动到底部
    modal.addEventListener('scroll', function() {
      const scrollTolerance = 5;
      if (modal.scrollTop + modal.clientHeight >= modal.scrollHeight - scrollTolerance) {
        checkbox.disabled = false;
      }
    });

    // 勾选后启用确认按钮
    checkbox.addEventListener('change', function() {
      confirmBtn.disabled = !checkbox.checked;
    });

    // 确认后记录状态并关闭弹窗
    confirmBtn.addEventListener('click', function() {
      localStorage.setItem('muse-tool-agreed', 'yes');
      localStorage.setItem('muse-tool-agreed-time', new Date().toISOString());
      modal.style.display = 'none';
      // 此处触发你的主功能初始化
    });
  })();
</script>
